import java.io.IOException;
import java.io.PrintWriter;
import javax.servlet.ServletException;
import javax.servlet.annotation.WebServlet;
import javax.servlet.http.HttpServlet;
import javax.servlet.http.HttpServletRequest;
import javax.servlet.http.HttpServletResponse;

@WebServlet("/sum")
public class SumServlet extends HttpServlet {

    @Override
    protected void doPost(HttpServletRequest req, HttpServletResponse res)
        throws ServletException, IOException {

        String numStr = req.getParameter("userNumber");
        int num = Integer.parseInt(numStr); // convert string to integer

        int sum = 0;
        for (int i = 1; i <= num; i++) {
            sum += i;
        }

        res.setContentType("text/html");
        PrintWriter out = res.getWriter();
        out.println("<html><body>");
        out.println("<h1>Sum from 1 to " + num + " is: " + sum + "</h1>");
        out.println("</body></html>");
    }
}

import java.io.IOException;
import java.io.PrintWriter;
import javax.servlet.ServletException;
import javax.servlet.annotation.WebServlet;
import javax.servlet.http.HttpServlet;
import javax.servlet.http.HttpServletRequest;
import javax.servlet.http.HttpServletResponse;

@WebServlet("/power")
public class PowerServlet extends HttpServlet {

    @Override
    protected void doPost(HttpServletRequest req, HttpServletResponse res)
        throws ServletException, IOException {

        int m = Integer.parseInt(req.getParameter("mNumber"));
        int n = Integer.parseInt(req.getParameter("nNumber"));

        int total = 0;
        for (int exp = 1; exp <= n; exp++) {
            int power = 1;
            for (int i = 1; i <= exp; i++) {
                power *= m;
            }
            total += power;
        }

        res.setContentType("text/html");
        PrintWriter out = res.getWriter();

        out.println("<html><body>");
        out.println("<h1>Computed:</h1>");
        out.println("<p>m = " + m + ", n = " + n + "</p>");
        out.println("<p><strong>Sum = " + total + "</strong></p>");
        out.println("</body></html>");
    }
}

import java.io.*;
import javax.servlet.*;
import javax.servlet.http.*;
import javax.servlet.annotation.WebServlet;

@WebServlet("/Computepattern")
public class ComputePattern extends HttpServlet {

    @Override
    protected void doPost(HttpServletRequest request, HttpServletResponse response)
            throws ServletException, IOException {

        String str1 = request.getParameter("m");
        String str2 = request.getParameter("n");

        int m = Integer.parseInt(str1);
        int n = Integer.parseInt(str2);
        int sum = 0;

        for (int i = 1; i <= n; i++) {
            int mult = 1;
            for (int j = 1; j <= i; j++) {
                mult *= m;
            }
            sum += mult;
        }

        response.setContentType("text/html");
        PrintWriter out = response.getWriter();
        out.println("<h2>Sum of powers: " + sum + "</h2>");
    }
}

import java.io.*;
import javax.servlet.*;
import javax.servlet.http.*;
import javax.servlet.annotation.WebServlet;

@WebServlet("/displaytime")
public class displaytime extends HttpServlet {

    @Override
    protected void doPost(HttpServletRequest request, HttpServletResponse response)
            throws ServletException, IOException {

        String minute = request.getParameter("num");
        int num = Integer.parseInt(minute);  
         
        int hours = num / 60;
        int remainingmin = num % 60;  
        
        response.setContentType("text/html");
        PrintWriter out = response.getWriter();
        out.println("<h2>Hours and Minutes: " + hours + " hours and " + remainingmin + " minutes</h2>");
    }
}
